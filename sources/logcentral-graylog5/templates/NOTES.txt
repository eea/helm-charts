1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "logcentral-graylog5.fullname" . }}-graylog-master)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "logcentral-graylog5.fullname" . }}-graylog-master'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "logcentral-graylog5.fullname" . }}-graylog-master --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "logcentral-graylog5.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=graylog-master" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:9000 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 9000:$CONTAINER_PORT
{{- end }}

2. Log service endpoints:
{{- if .Values.loadbalancer.enabled }}
   Syslog TCP/UDP: {{ include "logcentral-graylog5.fullname" . }}-loadbalancer:1514
   GELF TCP/UDP: {{ include "logcentral-graylog5.fullname" . }}-loadbalancer:12201
{{- end }}

3. IMPORTANT CONFIGURATION NOTES:
   - Set graylogMaster.config.passwordSecret to a random string of at least 16 characters
   - Set graylogMaster.config.rootPasswordSha2 to the SHA2 hash of your desired admin password
   - Set graylogMaster.config.httpExternalUri to your public Graylog URL
   - Configure external Elasticsearch connection via externalServices.elasticsearch.link
   - Configure Postfix SMTP credentials via postfix.config values

4. To generate a SHA2 password hash, run:
   echo -n "your-password" | sha256sum

5. Default admin username is "admin" - use the password you hashed for rootPasswordSha2
